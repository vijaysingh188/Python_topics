 
########create 3 table 
CREATE TABLE COUSTOMERS {
    coustomer_id int primary_key,
    coustomer_name varchar(255),
}


CREATE TABLE PRODUCTS{
    product_id int primary_key,
    product_name name varchar(255),
    product_price decimal(10,2),

}
CREATE TABLE ORDERS {
    orders_id int primary_key,
    orders_date date,
    coustomer_id int,
    product_id int, 
    foreignkey (coustomer_id) refereneces
    COUSTOMERS(coustomer_id),
    foreignkey (product_id) refereneces
    PRODUCTS(product_id),
}

############Join 3 table using inner join
SELECT ORDERS.orders_date,ORDERS.orders_id,
       COUSTOMERS.coustomer_name,
       PRODUCTS.product_name,PRODUCTS.product_price
FROM ORDERS 
INNER JOIN 
    COUSTOMERS ON ORDERS.coustomer_id = COUSTOMERS.coustomer_id
INNER JOIN 
    PRODUCTS ON ORDERS.product_id = PRODUCTS.product_id
